101 template < class iter_t, typename compare >
102 parallel_stable_sort_tag<iter_t,compare>::
103 parallel_stable_sort_tag ( range<iter_t> R, compare comp,
104  NThread NT ):NElem(0),Ptr(nullptr)
105 { //------------------------------- begin -----------------------------
106  assert ( R.valid());
107 
108  NElem = R.size() ;
109  size_t NPtr = ( NElem +1 )>>1 ;
110 
111  if ( NElem < NELEM_MIN or NT() == 1)
112  { spin_sort ( R.first , R.last, comp );
113  return ;
114  };
115  //------------------- check if sort --------------------------------------
116  bool SW = true ;
117  for ( iter_t it1 = R.first, it2 = R.first+1 ;
118  it2 != R.last and (SW = not comp(*it2,*it1));it1 = it2++);
119  if (SW) return ;
120 
121  Ptr = std::get_temporary_buffer<value_t>(NPtr).first ;
122  if ( Ptr == nullptr) throw std::bad_alloc();
123 
124  //---------------------------------------------------------------------
125  // Parallel Process
126  //---------------------------------------------------------------------
127  range<iter_t> R1 ( R.first, R.first + NPtr), R2(R.first + NPtr, R.last );
128  range <value_t*> Rbuf (Ptr , Ptr + NPtr );
129  sample_sort_tag<iter_t, compare> ( R1, comp,NT, Rbuf);
130  sample_sort_tag<iter_t, compare> ( R2, comp, NT,Rbuf);
131 
132  Rbuf = init_move (Rbuf, R1);
133  R = half_merge (R, Rbuf, R2, comp );
134 }; // end of constructor
