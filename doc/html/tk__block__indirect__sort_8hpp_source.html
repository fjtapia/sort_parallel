391 template < typename value_t >
392 void dispatcher<value_t>::exec (range_buf RB )
393 { //----------------------- begin -----------------------------------------
394  ptr_tkmono<value_t> ptkm ( nullptr) ;
395  ptr_tkdual<value_t> ptkd ( nullptr ) ;
396 
397  while ( not final())
398  { if ( dq_mono.pop_move_back( ptkm) )
399  { ptkm->exec(RB) ;
400  delete (ptkm.release()) ;
401  }
402  else
403  { if ( dq_dual.pop_move_back( ptkd) )
404  { uint32_t code =ptkd->exec( RB) ;
405  switch ( code)
406  { case 0: delete (ptkd.release()) ; break ;
407  case 1: dq_dual.emplace_back(std::move(ptkd)); break ;
408  case 2: dq_dual.emplace_front(std::move (ptkd));break ;
409  };
410  }

412  };
