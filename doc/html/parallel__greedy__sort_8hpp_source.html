293 { //------------------------- begin ----------------------

295

298  return ;
299  };
300 
301  //------------------- check if sort --------------------------------------
302  bool SW = true ;
303  for ( iter_t it1 = first, it2 = first+1 ;
304  it2 != last and (SW = not comp(*it2,*it1));it1 = it2++);
305  if (SW) return ;

307  //-------------------- pivoting ----------------------------------
308  pivot9 ( first, last ,comp);
309  const value_t & val = const_cast < value_t &>(* first);
310  iter_t c_first = first+1 , c_last = last-1;
311 
312  while ( comp(*c_first, val) ) ++c_first ;
313  while ( comp ( val,*c_last ) ) --c_last ;
314  while (not( c_first > c_last ))
315  { std::swap ( *(c_first++), *(c_last--));
316  while ( comp (*c_first, val) ) ++c_first;

318  }; // End while
