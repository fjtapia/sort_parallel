192 /// @param first : iterator to the first element of the range to sort
193 /// @param last : iterator after the last element to the range to sort
194 /// @param cmp : object for to compare two elements pointed by Iter_t iterators
195 /// @param num_thread : Number of threads to use in the process. When this value
196 /// is lower than 2, the sorting is done with 1 thread
197 /// @param paux : pointer to the auxiliary memory. If nullptr, the memory is
198 /// created inside the class
199 /// @param naux : number of elements of the memory pointed by paux
200 //-----------------------------------------------------------------------------
201 template <class Iter_t, typename Compare>
202 sample_sort<Iter_t, Compare>::sample_sort(Iter_t first, Iter_t last,
203  Compare cmp, uint32_t num_thread,
204  value_t *paux, size_t naux)
205  : nthread(num_thread), owner(false), comp(cmp), global_range(first, last),
206  global_buf(nullptr, nullptr)
207 { //-------------------------- begin -------------------------------------
208  assert((last - first) >= 0);
209  size_t nelem = size_t(last - first);
210  construct = false;
211  njob = 0;
212 
213  // Adjust when have many threads and only a few elements
214  while (nelem > thread_min and (nthread * nthread) > (nelem >> 3)) {
215  nthread /= 2;
216  };
217  ninterval = (nthread << 3);
218 
219  if (nthread < 2 or nelem <= (thread_min)) {
220  spin_sort<Iter_t, Compare>(first, last, comp);
221  return;
222  };
223 
224  //------------------- check if sort --------------------------------------
225  bool sw = true;
226  for (Iter_t it1 = first, it2 = first + 1;
227  it2 != last and (sw = not comp(*it2, *it1)); it1 = it2++)
228  ;
229 
230  if (sw) return;
231 
232  if (paux != nullptr) {
233  assert(naux != 0);
234  global_buf.first = paux;
235  global_buf.last = paux + naux;
236  owner = false;
237  }
238  else
239  {
240  value_t *ptr = std::get_temporary_buffer<value_t>(nelem).first;
241  if (ptr == nullptr) throw std::bad_alloc();
242  owner = true;
243  global_buf = range_buf(ptr, ptr + nelem);
244  };
245  //------------------------------------------------------------------------
246  // PROCESS
247  //------------------------------------------------------------------------
248  initial_configuration();
249  first_merge();
250  construct = true;
251  final_merge();
252 };
