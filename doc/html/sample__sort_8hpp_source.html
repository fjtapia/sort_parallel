317 template <class iter_t, typename compare>
318 void sample_sort_tag<iter_t,compare>::initial_configuration ( void)
319 { //--------------------------- begin --------------------------------------
320  std::vector <range_it> VMem_thread ;
321  std::vector <range_buf> VBuf_thread ;
322  size_t NElem = global_range.size() ;
323  //------------------------------------------------------------------------
324  size_t cupo = (NElem + NThr -1) / NThr ;
325  iter_t it_first = global_range.first ;
326  value_t * buf_first = global_buf.first ;
327 
328  for ( uint32_t i =0 ; i < NThr-1 ; ++i, it_first += cupo, buf_first += cupo)
329  { VMem_thread.emplace_back ( it_first, it_first + cupo) ;
330  VBuf_thread.emplace_back ( buf_first, buf_first + cupo );
331  };
332  VMem_thread.emplace_back ( it_first, global_range.last) ;
333  VBuf_thread.emplace_back ( buf_first, global_buf.last );
334 
335  //------------------------------------------------------------------------
336  // Sorting of the ranges
337  //------------------------------------------------------------------------
338  std::vector <std::future <void> > F ( NThr ) ;
339 
340  for ( uint32_t i =0 ; i < NThr ; ++i)
341  { F[i] = std::async (std::launch::async,
342  spin_sort<iter_t,value_t,compare>,
343  VMem_thread[i].first, VMem_thread[i].last,
344  comp, VBuf_thread[i].first,VBuf_thread[i].size());
345  };
346 
347  for ( uint32_t i =0 ; i < NThr ; ++i) F[i].get() ;
348 
349  //------------------------------------------------------------------------
350  // Obtain the vector of milestones
351  //------------------------------------------------------------------------
352  std::vector<iter_t> Vsample;
353  Vsample.reserve ( NThr * (Ninterval-1)) ;
354 
355  for ( uint32_t i =0 ; i < NThr ; ++i)
356  { size_t distance = VMem_thread[i].size() / Ninterval ;
357  for ( size_t j = 1 ,pos = distance; j < Ninterval; ++j,pos+=distance)
358  { Vsample.push_back (VMem_thread[i].first + pos );
359  };
360  };
361  typedef less_ptr_no_null <iter_t, compare> compare_ptr ;
362  spin_sort ( Vsample.begin() , Vsample.end(), compare_ptr(comp) );
363 
364  //------------------------------------------------------------------------
365  // Create the final milestone vector
366  //------------------------------------------------------------------------
367  std::vector<iter_t> Vmilestone ;
368  Vmilestone.reserve ( Ninterval);
369 
370  for ( uint32_t Pos =NThr >>1 ; Pos < Vsample.size() ; Pos += NThr )
371  Vmilestone.push_back ( Vsample [ Pos]);
372 
373  //------------------------------------------------------------------------
374  // Creation of the first vector of ranges
375  //------------------------------------------------------------------------
376  std::vector< std::vector<range <iter_t> > > VR (NThr);
377 
378  for ( uint32_t i =0 ; i < NThr; ++i)
379  { iter_t itaux = VMem_thread[i].first ;
380  for ( uint32_t k =0 ; k < (Ninterval -1) ; ++k)
381  { iter_t it2 = std::upper_bound ( itaux,
382  VMem_thread[i].last ,
383  * Vmilestone[k], comp );
384  VR[i].emplace_back ( itaux, it2);
385  itaux = it2 ;
386  };
387  VR[i].emplace_back(itaux,VMem_thread[i].last );
388  };
389 
390  //------------------------------------------------------------------------
391  // Copy in buffer and creation of the final matrix of ranges
392  //------------------------------------------------------------------------
393  VMem.resize ( Ninterval);
394  VBuf.resize ( Ninterval);
395  VMIni.reserve (Ninterval);
396  VBIni.reserve (Ninterval);
397 
398  for ( uint32_t i =0 ; i < Ninterval ; ++i)
399  { VMem[i].reserve ( NThr);
400  VBuf[i].reserve ( NThr);
401  };
402  iter_t it = global_range.first ;
403  value_t * it_buf = global_buf.first ;
404  for ( uint32_t k =0 ; k < Ninterval ; ++k)
405  { size_t N =0 ;
406  for ( uint32_t i = 0 ; i< NThr ; ++i)
407  { size_t N2 = VR[i][k].size();
408  if ( N2 != 0 ) VMem[k].push_back(VR[i][k] );
409  N += N2 ;
410  };
411  VMIni.emplace_back (it,it + N );
412  VBIni.emplace_back (it_buf , it_buf+ N) ;
413 
414  it += N ;
415  it_buf += N ;
416  };
