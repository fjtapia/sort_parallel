204 /// @brief constructor of the class
205 ///
206 /// @param first : iterator to the first element of the range to sort
207 /// @param last : iterator after the last element to the range to sort
208 /// @param cmp : object for to compare two elements pointed by Iter_t iterators
209 /// @param num_thread : Number of threads to use in the process. When this value
210 /// is lower than 2, the sorting is done with 1 thread
211 /// @param paux : pointer to the auxiliary memory. If nullptr, the memory is
212 /// created inside the class
213 /// @param naux : number of elements of the memory pointed by paux
214 //-----------------------------------------------------------------------------
215 template < class Iter_t, typename Compare >
216 sample_sort< Iter_t, Compare >::sample_sort( Iter_t first, Iter_t last,
217  Compare cmp, uint32_t num_thread,
218  value_t *paux, size_t naux )
219  : nthread( num_thread ), owner( false ), comp( cmp ),
220  global_range( first, last ), global_buf( nullptr, nullptr ),
221  error( false )
222 { //-------------------------- begin -------------------------------------
223  assert( ( last - first ) >= 0 );
224  size_t nelem = size_t( last - first );
225  construct = false;
226  njob = 0;
227  vfuture.resize( nthread );
228 
229  // Adjust when have many threads and only a few elements
230  while ( nelem > thread_min and ( nthread * nthread ) > ( nelem >> 3 ) ) {
231  nthread /= 2;
232  };
233  ninterval = ( nthread << 3 );
234 
235  if ( nthread < 2 or nelem <= ( thread_min ) ) {
236  spin_sort< Iter_t, Compare >( first, last, comp );
237  return;
238  };
239 
240  //------------------- check if sort --------------------------------------
241  bool sw = true;
242  for ( Iter_t it1 = first, it2 = first + 1;
243  it2 != last and ( sw = not comp( *it2, *it1 ) ); it1 = it2++ );
244  if ( sw ) return;
245 
246  if ( paux != nullptr ) {
247  assert( naux != 0 );
248  global_buf.first = paux;
249  global_buf.last = paux + naux;
250  owner = false;
251  }
252  else
253  {
254  value_t *ptr = std::get_temporary_buffer< value_t >( nelem ).first;
255  if ( ptr == nullptr ) throw std::bad_alloc();
256  owner = true;
257  global_buf = range_buf( ptr, ptr + nelem );
258  };
259  //------------------------------------------------------------------------
260  // PROCESS
261  //------------------------------------------------------------------------
262  try
263  {
264  initial_configuration();
265  }
266  catch ( std::bad_alloc & )
267  {
268  error = true;
269  };
270  if ( not error ) {
271  first_merge();
272  construct = true;
273  final_merge();
274  };
275  if ( error ) {
276  destroy_all();
277  throw std::bad_alloc();
278  };
279 };

315  vbuf_thread.reserve( nthread + 1 );
316 
317  for ( uint32_t i = 0; i < ( nthread - 1 );
318  ++i, it_first += cupo, buf_first += cupo )
319  {
320  vmem_thread.emplace_back( it_first, it_first + cupo );
321  vbuf_thread.emplace_back( buf_first, buf_first + cupo );
322  };
323 
324  vmem_thread.emplace_back( it_first, global_range.last );
325  vbuf_thread.emplace_back( buf_first, global_buf.last );
326 
327  //------------------------------------------------------------------------
328  // Sorting of the ranges
329  //------------------------------------------------------------------------
330  std::vector< std::future< void > > vfuture( nthread );
331 
332  for ( uint32_t i = 0; i < nthread; ++i ) {
333  auto func = [=]() {
334  spin_sort_t( vmem_thread[ i ].first, vmem_thread[ i ].last, comp,
335  vbuf_thread[ i ] );
336  };
337  vfuture[ i ] = std::async( std::launch::async, func );
338  };
339 
340  for ( uint32_t i = 0; i < nthread; ++i ) vfuture[ i ].get();
341 
342  //------------------------------------------------------------------------
343  // Obtain the vector of milestones
344  //------------------------------------------------------------------------
345  std::vector< Iter_t > vsample;
346  vsample.reserve( nthread * ( ninterval - 1 ) );
347 
348  for ( uint32_t i = 0; i < nthread; ++i ) {
349  size_t distance = vmem_thread[ i ].size() / ninterval;
350  for ( size_t j = 1, pos = distance; j < ninterval;
351  ++j, pos += distance )
352  {
353  vsample.push_back( vmem_thread[ i ].first + pos );
354  };
355  };

357  typedef typename std::vector< Iter_t >::iterator it_to_it;
358 
359  spin_sort< it_to_it, compare_ptr >( vsample.begin(), vsample.end(),
360  compare_ptr( comp ) );
361 
362  //------------------------------------------------------------------------
363  // Create the final milestone vector
364  //------------------------------------------------------------------------
365  std::vector< Iter_t > vmilestone;
366  vmilestone.reserve( ninterval );
367 
368  for ( uint32_t pos = nthread >> 1; pos < vsample.size(); pos += nthread ) {
369  vmilestone.push_back( vsample[ pos ] );
370  };
371 
372  //------------------------------------------------------------------------
373  // Creation of the first vector of ranges
374  //------------------------------------------------------------------------
375  std::vector< std::vector< range< Iter_t > > > vv_range_first( nthread );
376 
377  for ( uint32_t i = 0; i < nthread; ++i ) {
378  Iter_t itaux = vmem_thread[ i ].first;
379 
380  for ( uint32_t k = 0; k < ( ninterval - 1 ); ++k ) {
381  Iter_t it2 = std::upper_bound( itaux, vmem_thread[ i ].last,
382  *vmilestone[ k ], comp );
383 
384  vv_range_first[ i ].emplace_back( itaux, it2 );
385  itaux = it2;
386  };
387  vv_range_first[ i ].emplace_back( itaux, vmem_thread[ i ].last );
388  };
389 
390  //------------------------------------------------------------------------
391  // Copy in buffer and creation of the final matrix of ranges
392  //------------------------------------------------------------------------
393  vv_range_it.resize( ninterval );
394  vv_range_buf.resize( ninterval );
395  vrange_it_ini.reserve( ninterval );
396  vrange_buf_ini.reserve( ninterval );
397 
398  for ( uint32_t i = 0; i < ninterval; ++i ) {
399  vv_range_it[ i ].reserve( nthread );
400  vv_range_buf[ i ].reserve( nthread );
401  };
402 
403  Iter_t it = global_range.first;
404  value_t *it_buf = global_buf.first;
405 
406  for ( uint32_t k = 0; k < ninterval; ++k ) {
407  size_t nelem_interval = 0;
408 
409  for ( uint32_t i = 0; i < nthread; ++i ) {
410  size_t nelem_range = vv_range_first[ i ][ k ].size();
411  if ( nelem_range != 0 ) {
412  vv_range_it[ k ].push_back( vv_range_first[ i ][ k ] );
413  };
414  nelem_interval += nelem_range;
415  };
416 
417  vrange_it_ini.emplace_back( it, it + nelem_interval );
418  vrange_buf_ini.emplace_back( it_buf, it_buf + nelem_interval );
419 
420  it += nelem_interval;
421  it_buf += nelem_interval;
422  };
