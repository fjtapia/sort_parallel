29 /// @struct block_pos
30 /// @brief represent a pair of values, a position represented as an unsigned
31 /// variable ( position ), and a bool variable ( side ). They are packed
32 /// in a size_t variable. The Least Significant Bit is the bool variable,
33 /// and the others bits are the position
34 //----------------------------------------------------------------------------
35 class block_pos
36 {
37  //------------------------------ VARIABLES -----------------------------
38  size_t num; // number which store a position and a bool side
39 
40  public:
41  //----------------------------- FUNCTIONS ------------------------------
42  block_pos(void) : num(0){};
43  //
44  //-------------------------------------------------------------------------
45  // function : block_pos
46  /// @brief constructor from a position and a side
47  /// @param position : position to sotre
48  /// @param side : side to store
49  //-------------------------------------------------------------------------
50  block_pos(size_t position, bool side = false)
51  {
52  num = (position << 1) + ((side) ? 1 : 0);
53  };
54  //
55  //-------------------------------------------------------------------------
56  // function : pos
57  /// @brief obtain the position stored inside the block_pos
58  /// @return position
59  //-------------------------------------------------------------------------
60  size_t pos(void) const { return (num >> 1); };
61  //
62  //-------------------------------------------------------------------------
63  // function : pos
64  /// @brief store a position inside the block_pos
65  /// @param position : value to store
66  //-------------------------------------------------------------------------
67  void pos(size_t position) { num = (position << 1) + (num & 1); };
68  //
69  //-------------------------------------------------------------------------
70  // function : side
71  /// @brief obtain the side stored inside the block_pos
72  /// @return bool value
73  //-------------------------------------------------------------------------
74  bool side(void) const { return ((num & 1) != 0); };
75  //
76  //-------------------------------------------------------------------------
77  // function : side
78  /// @brief store a bool value the block_pos
79  /// @param sd : bool value to store
80  //-------------------------------------------------------------------------
81  void side(bool sd) { num = (num & ~1) + ((sd) ? 1 : 0); };
82 
83 }; // end struct block_pos
