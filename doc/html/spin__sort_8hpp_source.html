43 /// @brief this function divide r_input in two parts, sort it,and merge moving
44 /// the elements to range_buf
45 /// @param range_input : range with the elements to sort
46 /// @param range_buffer : range with the elements sorted
47 /// @param comp : object for to compare two elements
48 /// @param level : when is 1, sort with the insertion_sort algorithm
49 /// if not make a recursive call splitting the ranges
50 //-----------------------------------------------------------------------------
51 template <class Iter1_t, class Iter2_t, class Compare>
52 void range_sort(const range<Iter1_t> &range_input,
53  const range<Iter2_t> &range_buffer, Compare comp,
54  uint32_t level)
55 { //---------------------------- begin ------------------------------------
56  typedef range<Iter1_t> range_it1;
57  typedef range<Iter2_t> range_it2;
58  assert(range_input.size() == range_buffer.size() and level != 0);
59 
60  size_t nelem1 = (range_input.size() + 1) >> 1;
61  range_it1 range_input1(range_input.first, range_input.first + nelem1),
62  range_input2(range_input.first + nelem1, range_input.last);
63 
64  if (level < 2) {
65  insertion_sort(range_input1.first, range_input1.last, comp);
66  insertion_sort(range_input2.first, range_input2.last, comp);
67  }
68  else
69  {
70  range_sort(range_it2(range_buffer.first, range_buffer.first + nelem1),
71  range_input1, comp, level - 1);
72 
73  range_sort(range_it2(range_buffer.first + nelem1, range_buffer.last),
74  range_input2, comp, level - 1);
75  };
76  full_merge(range_buffer, range_input1, range_input2, comp);
77 };

158 template <class Iter_t, class Compare>
159 spin_sort<Iter_t, Compare>::spin_sort(Iter_t first, Iter_t last, Compare comp,
160  value_t *paux, size_t naux)
161  : ptr(paux), nptr(naux), construct(false), owner(false)
162 { //-------------------------- begin -------------------------------------
163  range<Iter_t> range_input(first, last);
164  assert(range_input.valid());
165 
166  size_t nelem = range_input.size();
167  owner = construct = false;
168 
169  nptr = (nelem + 1) >> 1;
170  size_t nelem_1 = nptr;
171  size_t nelem_2 = nelem - nelem_1;
172 
173  if (nelem <= (Sort_min << 1)) {
174  insertion_sort(range_input.first, range_input.last, comp);
175  return;
176  };
177  //------------------- check if sort --------------------------------------
178  bool sw = true;
179  for (Iter_t it1 = range_input.first, it2 = range_input.first + 1;
180  it2 != range_input.last and (sw = not comp(*it2, *it1)); it1 = it2++)
181  ;
182  if (sw) return;
183 
184  if (ptr == nullptr) {
185  ptr = std::get_temporary_buffer<value_t>(nptr).first;
186  if (ptr == nullptr) throw std::bad_alloc();
187  owner = true;
188  };
189  range_buf range_aux(ptr, (ptr + nptr));
190 
191  //------------------------------------------------------------------------
192  // Process
193  //------------------------------------------------------------------------
194  uint32_t nlevel = nbits64(((nelem + Sort_min - 1) / Sort_min) - 1) - 1;
195  assert(nlevel != 0);
196 
197  if ((nlevel & 1) == 1) {
198  //---------------------------------------------------------------------
199  // if the number of levels is odd, the data are in the first parameter
200  // of range_sort, and the results appear in the second parameter
201  //---------------------------------------------------------------------
202  range_it range_1(first, first + nelem_2),
203  range_2(first + nelem_2, last);
204  range_aux = uninit_move(range_aux, range_2);
205  construct = true;
206 
207  range_sort(range_aux, range_2, comp, nlevel);
208  range_buf rng_bx(range_aux.first, range_aux.first + nelem_2);
209 
210  range_sort(range_1, rng_bx, comp, nlevel);
211  half_merge(range_input, rng_bx, range_2, comp);
212  }
213  else
214  {
215  //--------------------------------------------------------------------
216  // If the number of levels is even, the data are in the second
217  // parameter of range_sort, and the results are in the same parameter
218  //---------------------------------------------------------------------
219  range_it range_1(first, first + nelem_1),
220  range_2(first + nelem_1, last);
221  range_aux = uninit_move(range_aux, range_1);
222  construct = true;
223 
224  range_sort(range_1, range_aux, comp, nlevel);
225 
226  range_1.last = range_1.first + range_2.size();
227  range_sort(range_1, range_2, comp, nlevel);
228  half_merge(range_input, range_aux, range_2, comp);
229  };
230 };
