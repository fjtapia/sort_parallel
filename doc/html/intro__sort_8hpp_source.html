39 /// @brief : return the iterator to the mid value of the three values passsed
40 /// as parameters
41 //
42 /// @param iter_1 : iterator to the first value
43 /// @param iter_2 : iterator to the second value
44 /// @param iter_3 : iterator to the third value
45 /// @param comp : object for to compare two values
46 /// @return iterator to the mid value
47 //-----------------------------------------------------------------------------
48 template < typename Iter_t, typename Compare >
49 inline Iter_t mid3( Iter_t iter_1, Iter_t iter_2, Iter_t iter_3, Compare comp )
50 {
51  return comp( *iter_1, *iter_2 )
52  ? ( comp( *iter_2, *iter_3 )
53  ? iter_2
54  : ( comp( *iter_1, *iter_3 ) ? iter_3 : iter_1 ) )
55  : ( comp( *iter_3, *iter_2 )
56  ? iter_2
57  : ( comp( *iter_3, *iter_1 ) ? iter_3 : iter_1 ) );
58 };

134 template < class Iter_t, typename Compare >
135 void intro_sort_internal( Iter_t first, Iter_t last, uint32_t level,
136  Compare comp )
137 { //------------------------------ begin -----------------------------------
138  typedef typename std::iterator_traits< Iter_t >::value_type value_t;
139 
140  const uint32_t nmin = 32;
141  size_t nelem = last - first;
142  if ( nelem < nmin ) return insertion_sort( first, last, comp );
143 
144  if ( level == 0 ) {
145  heap_sort< Iter_t, Compare >( first, last, comp );
146  return;
147  };
148 
149  //--------------------- division ----------------------------------
150  pivot3( first, last, comp );
151 
152  const value_t &val = const_cast< value_t & >( *first );
153  Iter_t c_first = first + 1, c_last = last - 1;
154 
155  while ( comp( *c_first, val ) ) ++c_first;
156  while ( comp( val, *c_last ) ) --c_last;
157 
158  while ( not( c_first > c_last ) ) {
159  std::swap( *( c_first++ ), *( c_last-- ) );
160  while ( comp( *c_first, val ) ) ++c_first;
161  while ( comp( val, *c_last ) ) --c_last;
162  }; // End while
163
