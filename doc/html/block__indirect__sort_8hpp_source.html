159 /// @brief begin with the execution of the functions stored in works
160 /// @param first : iterator to the first element of the range to sort
161 /// @param last : iterator after the last element to the range to sort
162 /// @param comp : object for to compare two elements pointed by Iter_t
163 /// iterators
164 /// @param nthr : Number of threads to use in the process.When this value
165 /// is lower than 2, the sorting is done with 1 thread
166 //-------------------------------------------------------------------------
167 template < uint32_t Block_size, uint32_t Group_size, class Iter_t,
168  class Compare >
169 block_indirect_sort< Block_size, Group_size, Iter_t,
170  Compare >::block_indirect_sort (Iter_t first, Iter_t last,
171  Compare cmp, uint32_t nthr)
172  : bk (first, last, cmp), counter (0), ptr (nullptr), construct (false),
173  nthread (nthr)
174 {
175  try
176  {
177  assert ((last - first) >= 0);
178  size_t nelem = size_t (last - first);
179  if (nelem == 0) return;
180 
181  //------------------- check if sort -----------------------------------
182  bool sorted = true;
183  for (Iter_t it1 = first, it2 = first + 1;
184  it2 != last and (sorted = not bk.cmp (*it2, *it1)); it1 = it2++)
185  ;
186  if (sorted) return;
187 
188  //---------------- check if only single thread -----------------------
189  size_t nthreadmax = nelem / (Block_size * Group_size) + 1;
190  if (nthread > nthreadmax) nthread = (uint32_t)nthreadmax;
191  uint32_t nbits_size = (util::nbits64 (sizeof (value_t))) >> 1;
192  if (nbits_size > 5) nbits_size = 5;
193  size_t max_per_thread = 1 << (18 - nbits_size);
194 
195  if (nelem < (max_per_thread) or nthread < 2) {
196  intro_sort (first, last, bk.cmp);
197  return;
198  };
199 
200  //----------- creation of the temporary buffer --------------------
201  ptr = std::get_temporary_buffer< value_t > (Block_size * nthread).first;
202  if (ptr == nullptr) {
203  bk.error = true;
204  throw std::bad_alloc ( );
205  };
206 
207  rglobal_buf = range_buf (ptr, ptr + (Block_size * nthread));
208  util::init (rglobal_buf, *first);
209  construct = true;
210 
211  // creation of the buffers for the threads
212  std::vector< value_t * > vbuf (nthread);
213  for (uint32_t i = 0; i < nthread; ++i) {
214  vbuf[i] = ptr + (i * Block_size);
215  };
216 
217  // Insert the first work in the stack
218  util::atomic_write (counter, 1);
219  function_t f1 = [&]( ) {
220  start_function ( );
221  util::atomic_sub (counter, 1);
222  };
223  bk.works.emplace_back (f1);
224 
225  //---------------------------------------------------------------------
226  // PROCESS
227  //---------------------------------------------------------------------
228  std::vector< std::future< void > > vfuture (nthread);
229 
230  // The function launched with the futures is "execute the functions of
231  // the stack until this->counter is zero
232  // vbuf[i] is the memory from the main thread for to configure the
233  // thread local buffer
234  for (uint32_t i = 0; i < nthread; ++i) {
235  auto f1 = [=, &vbuf]( ) { bk.exec (vbuf[i], this->counter); };
236  vfuture[i] = std::async (std::launch::async, f1);
237  };
238  for (uint32_t i = 0; i < nthread; ++i) vfuture[i].get ( );
239  if (bk.error) throw std::bad_alloc ( );
240  }
241  catch (std::bad_alloc &)
242  {
243  destroy_all ( );
244  throw;
245  }
246 };

250 // function : split_rage
251 /// @brief this function splits a range of positions in the index, and
252 /// depending of the size, sort directly or make to a recursive call
253 /// to split_range
254 /// @param pos_index1 : first position in the index
255 /// @param pos_index2 : position after the last in the index
256 /// @param level_thread : depth of the call. When 0 sort the blocks
257 //-----------------------------------------------------------------------------
258 template < uint32_t Block_size, uint32_t Group_size, class Iter_t,
259  class Compare >
260 void block_indirect_sort< Block_size, Group_size, Iter_t,
261  Compare >::split_range (size_t pos_index1,
262  size_t pos_index2,
263  uint32_t level_thread)
264 {
265  size_t nblock = pos_index2 - pos_index1;
266 
267  //-------------------------------------------------------------------------
268  // In the blocks not sorted, the physical position is the logical position
269  //-------------------------------------------------------------------------
270  Iter_t first = bk.get_block (pos_index1).first;
271  Iter_t last = bk.get_range (pos_index2 - 1).last;
272 
273  if (nblock < Group_size) {
274  intro_sort (first, last, bk.cmp);
275  return;
276  };
277 
278  size_t pos_index_mid = pos_index1 + (nblock >> 1);
279  atomic_t son_counter (1);
280 
281  //-------------------------------------------------------------------------
282  // Insert in the stack the work for the second part, and the actual thread,
283  // execute the first part
284  //-------------------------------------------------------------------------
285  if (level_thread != 0) {
286  auto f1 = [=, &son_counter]( ) {
287  split_range (pos_index_mid, pos_index2, level_thread - 1);
288  util::atomic_sub (son_counter, 1);
289  };
290  bk.works.emplace_back (f1);
291  if (bk.error) return;
292  split_range (pos_index1, pos_index_mid, level_thread - 1);
293  }
294  else
295  {
296  Iter_t mid = first + ((nblock >> 1) * Block_size);
297  auto f1 = [=, &son_counter]( ) {
298  parallel_sort_t (bk, mid, last);
299  util::atomic_sub (son_counter, 1);
300  };
301  bk.works.emplace_back (f1);
302  if (bk.error) return;
303  parallel_sort_t (bk, first, mid);
304  };
305  bk.exec (son_counter);
306  if (bk.error) return;
307  merge_blocks_t (bk, pos_index1, pos_index_mid, pos_index2);
308 };

312 /// @brief this function init the process. When the number of threads is lower
313 /// than a predefined value, sort the elements with a parallel introsort.
314 //-----------------------------------------------------------------------------
315 template < uint32_t Block_size, uint32_t Group_size, class Iter_t,
316  class Compare >
317 void block_indirect_sort< Block_size, Group_size, Iter_t,
318  Compare >::start_function (void)
319 {
320  if (nthread < BOOST_NTHREAD_BORDER) {
321  parallel_sort_t (bk, bk.global_range.first, bk.global_range.last);
322  }
323  else
324  {
325  size_t level_thread = nbits64 (nthread - 1);
326  split_range (0, bk.nblock, level_thread - 1);
327  if (bk.error) return;
328  move_blocks_t k (bk);
329  };
