236 // function : block_indirect_sort
237 /// @brief constructor of the class
238 /// @param [in] R1 : range of iterators to the elements to sot
239 /// @param [in] comp : object for to compare two elements
240 /// @param [in] NT : NThread object for to define the number of threads to use
241 /// in the process. By default is the number of thread HW
242 //-----------------------------------------------------------------------------
243 template < uint32_t BLK_SIZE,uint32_t RNG_SIZE, class iter_type,class compare>
244 block_indirect_sort<BLK_SIZE, RNG_SIZE, iter_type,compare>::
245 block_indirect_sort ( range_it R1, compare cmp, NThread NT )
246  :R(R1),Rtail (R1), Ptr (nullptr) ,comp( cmp),
247  NB(0),construct (false)
248 { //-------------------------- begin -------------------------------------
249  assert ( R.valid() );
250  size_t NElem = R.size() ;
251 
252  if ( NT() <2 or NElem <= ( Thread_min))
253  { intro_sort (R.first, R.last, comp);
254  return ;
255  };
256  LevelNT = NBits64( NT()-1 ) ;
257 
258  //------------------- check if sort --------------------------------------
259  bool SW = true ;
260  for ( iter_t it1 = R.first, it2 = R.first+1 ;
261  it2 != R.last and (SW = not comp(*it2,*it1));it1 = it2++);
262  if (SW) return ;
263 
264  //-------------- MaxThread and LevelThread ----------------------
265  uint32_t NBSize = (NBits64(sizeof (value_t)))>>1;
266  if ( NBSize > 5 ) NBSize = 5 ;
267  MaxThread = 1<< (18 - NBSize);
268  LevelThread = (NBits64 ( MaxThread) <<1);
269 
270  //----------------- creation of the temporary buffer --------------------
271  Ptr =std::get_temporary_buffer<value_t>( BLK_SIZE*NT()).first ;
272  if ( Ptr == nullptr) throw std::bad_alloc() ;
273  Rglobal_buf = range_buf ( Ptr , Ptr + (BLK_SIZE *NT()) );
274  bs_util::init ( Rglobal_buf, *R.first);
275  construct = true ;
276 
277  // --------------- create the ranges for the futures -------------------
278  std::vector< range_buf> VRB;
279  VRB.reserve ( NT() ) ;
280  value_t * P1 = Ptr ;
281  for ( uint32_t i =0 ;i < NT() ; ++i, P1 += BLK_SIZE )
282  VRB.emplace_back ( P1, P1 + BLK_SIZE ) ;
283 
284  // -------------------- create and fill Vpos ----------------------------
285  NB = NElem / BLK_SIZE ;
286  V.reserve ( NB +1 ) ;
287  for ( size_t i =0 ; i <NB ; ++i) V.emplace_back ( i,true);
288  Rtail.first = R.first + (NB * BLK_SIZE);
289  if ( (NElem % BLK_SIZE) != 0 ) V.emplace_back ( NB, true);
290  NB = V.size() ;
291 
292  //----------------------------------------------------------------------
293  atomic_write ( D.N, 1);
294  D.dq_dual.emplace_back ( new tk_global_t ( *this));
295 
296  //------------------------------------------------------------------------
297  // PROCESS
298  //------------------------------------------------------------------------
299  std::vector <std::future <void> > F ( NT() ) ;
300 
301  for ( uint32_t i =0 ; i < NT() ; ++i)
302  F[i] = std::async ( std::launch::async ,
303  &dispatcher<value_t>::exec, &D,VRB[i] );
304 
305  for ( uint32_t i =0 ; i < NT() ; ++i) F[i].get() ;
306 };

309 // function : split_rage
310 /// @brief this function splits a range, and depending of the size, sort
311 /// directly or call to a recursive split_range
312 /// @param [in] pf : pointer to the atomic counter of the father
313 /// @param [in] RP : range of positions to split
314 /// @param [in] Level : depth of the split
315 //-----------------------------------------------------------------------------
316 template < uint32_t BLK_SIZE, uint32_t RNG_SIZE,
317  class iter_type, typename compare >
318 void block_indirect_sort<BLK_SIZE, RNG_SIZE, iter_type,compare>::
319 split_range ( atomic_t *pf ,range_pos RP, uint32_t Level )
320 { //----------------------------- begin -----------------------------------
321  size_t NB1 = (RP.size() )>>1 ;
322  range_pos RP1 ( RP.first,RP.first + NB1 );
323  range_pos RP2 ( RP.first + NB1 , RP.last);
324  bs_tools::atomic_write ( *pf, 2);
325  if ( Level == 0 )
326  { iter_t it1 = blk_pos ( RP1.first);
327  iter_t it2 = it1 + (NB1 * BLK_SIZE) ;
328  iter_t it3 = rng_pos ( RP2.last-1).last;
329  D.dq_mono.emplace_back ( new tk_sort_t ( pf,*this, it1, it2, Level -1));
330  D.dq_mono.emplace_back ( new tk_sort_t ( pf,*this, it2, it3, Level -1));
331  }
332  else
333  { D.dq_dual.emplace_back (new tk_split_t ( pf, *this, RP1, Level -1) );
334  D.dq_dual.emplace_back (new tk_split_t ( pf, *this, RP2, Level -1) );
335  };
336 };

339 // function : merge_range
340 /// @brief make the indirect merge of the two range_pos inside RP
341 /// @param [in] pf : pointer to the atomic counter of the father
342 /// @param [in] RP : rnage_pos which contains the two ranges to merge
343 /// @param [in] Rbuf : auxiliary memory for to do the merge
344 //-----------------------------------------------------------------------------
345 template < uint32_t BLK_SIZE, uint32_t RNG_SIZE,
346  class iter_type, typename compare >
347 void block_indirect_sort<BLK_SIZE, RNG_SIZE, iter_type,compare>::
348 merge_range (atomic_t *pf , range_pos RP , range_buf Rbuf)
349 { //------------------------ begin ------------------------------
350  size_t NB1 = (RP.size() )>>1 ;
351  range_pos RP1 ( RP.first,RP.first + NB1 );
352  range_pos RP2 ( RP.first + NB1 , RP.last);
353 
354  //-----------------------------------------------------------------------
355  // Merging of the two intervals
356  //-----------------------------------------------------------------------
357  std::vector<blk> Vb1, Vb2 ;
358  Vb1.reserve ( RP1.size() );
359  Vb2.reserve ( RP2.size() );
360 
361  for ( size_t i = RP1.first ; i < RP1.last ; ++i)
362  Vb1.emplace_back( V[i].pos, true) ;
363  for ( size_t i = RP2.first; i < RP2.last ; ++i)
364  Vb2.emplace_back( V[i].pos, false) ;
365 
366  // tail process
367  bool Btail = ( Vb2.back().pos == (NB -1) and Rtail.first != Rtail.last);
368  if ( Btail)
369  { Vb2.pop_back();
370  size_t posback = V[RP1.last - 1].pos ;
371  range_it rback = rng_pos (posback );
372  if ( not in_place_merge_uncontiguous ( rback, Rtail, Rbuf, comp) )
373  { iter_t itaux = blk_pos(V[RP1.last -2].pos )+ (BLK_SIZE -1);
374  if ( comp ( *rback.first , *itaux) )
375  { Vb2.emplace_back( Vb1.back());
376  Vb1.pop_back();
377  };
378  };
379  };
380 
381  compare_blk<BLK_SIZE, iter_t ,compare> cmp_blk ( R.first, comp) ;
382 
383  bs_util::lwl::full_merge (Vb1.begin() , Vb1.end(),
384  Vb2.begin() , Vb2.end(),
385  V.begin() + RP.first, cmp_blk);
386 
387  //------------------------------------------------------------------------
388  // Extracting the ranges for to merge the elements
389  //------------------------------------------------------------------------
390  std::vector <range_pos> Vrp ;
391  Vrp.reserve ( RP.size() / RNG_SIZE +1);
392  if ( RNG_SIZE >= RP.size() )
393  { Vrp.emplace_back ( RP);
394  return ;
395  };
396  size_t PosIni = RP.first ;
397  size_t PosLast = (Btail)? RP.last -1:RP.last ;
398  while ( PosIni < PosLast)
399  { size_t Pos = PosIni + RNG_SIZE ;
400  while ( Pos < PosLast and V[Pos-1].side == V[Pos].side) ++Pos ;
401  if ( Pos < PosLast)
402  in_place_merge_uncontiguous ( rng_pos ( V[Pos-1].pos),
403  rng_pos (V[Pos].pos), Rbuf, comp);
404  else Pos = PosLast ;
405  Vrp.emplace_back( PosIni, Pos);
406  PosIni = Pos ;
407  };
408 
409  bs_tools::atomic_write ( *pf, Vrp.size());
410  for ( size_t i = 0 ; i < Vrp.size(); ++i)
411  { D.dq_mono.emplace_back(new tk_merge_t ( pf,*this, Vrp[i]));
412  };
413 };

416 // function : sort_blocks
417 /// @brief generate the vectors with the movements , needed for to move the
418 /// elements according with the index V
419 /// @param [in] pf : pointer to the atomic counter of the father
420 //-----------------------------------------------------------------------------
421 template < uint32_t BLK_SIZE, uint32_t RNG_SIZE,
422  class iter_type, typename compare >
423 void block_indirect_sort<BLK_SIZE, RNG_SIZE, iter_type,compare>::
424 sort_blocks (atomic_t *pf)
425 { //------------------------------------------------------------------------
426  // sort blocks
427  //------------------------------------------------------------------------
428  std::vector<size_t> Vaux ;
429  size_t pos_ini =0 , pos_src =0, pos_dest =0 ;
430  while ( pos_ini < V.size() )
431  { while (pos_ini < V.size() and V[pos_ini].pos == pos_ini) ++pos_ini;
432  if ( pos_ini == V.size() ) break ;
433  Vaux.clear() ;
434  pos_src = pos_dest = pos_ini ;
435  Vaux.push_back (pos_ini);
436  while ( V[ pos_dest].pos != pos_ini )
437  { pos_src = V[ pos_dest].pos;
438  Vaux.push_back ( pos_src) ;
439  V[ pos_dest].pos = pos_dest;
440  pos_dest = pos_src ;
441  };
442  V[ pos_dest].pos = pos_dest;
443  Vpos.push_back( Vaux);
444  };
445  bs_tools::atomic_write ( *pf, Vpos.size() );
446  dq_tkmono <value_t> dqm ;
447  dq_tkdual <value_t> dqd ;
448  for ( size_t i =0 ; i < Vpos.size() ; ++i)
449  { if ( Vpos[i].size() < MV_SIZE )
450  dqm.emplace_back( new tk_simple_move_t ( pf,*this, Vpos[i].begin(),
451  Vpos[i].end() ));
452  else
453  dqd.emplace_back( new tk_move_t ( pf ,*this , Vpos[i].begin() ,
454  Vpos[i].end()));
455  };
456  if ( dqm.size() != 0 ) D.dq_mono.push_back (std::move (dqm));
457  if ( dqd.size() != 0 ) D.dq_dual.push_back (std::move( dqd));
458 
459 };

482 // function : move_blocks
483 /// @brief split the long movement sequence in parts and create objects
484 /// tk_simple_move_t and insert in the deque
485 /// @param [in] pf : pointer to the atomic counter of the father
486 /// @param [in] first : first position of the block to move
487 /// @param [in] last : last position of the block to move
488 /// @param [in] Vseq : When the sequence is very long, is divided in parts,
489 /// and appear a new sequence , which must be done when all
490 /// the others are finished. This sequence is Vseq
491 /// @param [in] Rbuf : auxiliary memory for to do the movements
492 /// @return 0 : all is done, Vseq empty 1 : long sequence
493 /// @remarks
494 //-----------------------------------------------------------------------------
495 template < uint32_t BLK_SIZE, uint32_t RNG_SIZE,
496  class iter_type, typename compare >
497 uint32_t block_indirect_sort<BLK_SIZE, RNG_SIZE, iter_type,compare>::
498 move_blocks ( atomic_t *pf ,itpos_t first, itpos_t last,
499  std::vector<size_t> &Vseq , range_buf Rbuf )
500 { //----------------------- begin ----------------------------------------
501  Vseq.clear() ;
502  bs_tools::atomic_write ( *pf, 0);
503  if ( first == last ) return 0;
504  size_t NElem = last - first ;
505  if ( NElem <= MV_SIZE )
506  { move_simple_blocks( first, last,Rbuf) ;
507  return 0 ;
508  };
509  size_t NB = NElem / MV_SIZE ;
510  if ( (NElem % MV_SIZE )!= 0 ) NB++ ;
511 
512  //std::vector<size_t> Vseq ;
513  Vseq.reserve ( NB);
514  itpos_t it1 = first ;
515  bs_tools::atomic_write ( *pf, NB);
516  for ( size_t i =0 ; i < NB ; ++i)
517  { itpos_t it2 = ( i == (NB-1))? last:it1+MV_SIZE ;
518  Vseq.push_back (*(it2-1) );
519  D.dq_mono.emplace_back( new tk_simple_move_t ( pf, *this, it1, it2));
520  it1 = it2 ;
521  };
522  return 1 ;
523 };

526 // function : sort_thread
527 /// @brief this function is the work asigned to each thread in the parallel
528 /// process
529 /// @param [in] first : iterator to the first element to sort
530 /// @param [in] last : iterator to the next element after the last
531 /// @param [in] level : level of depth
532 //------------------------------------------------------------------------
533 template < uint32_t BLK_SIZE, uint32_t RNG_SIZE,
534  class iter_type, typename compare >
535 void block_indirect_sort<BLK_SIZE, RNG_SIZE, iter_type,compare>::
536 sort_thread ( atomic_t *pf, iter_t RESTRICT first,
537  iter_t RESTRICT last , uint32_t level )
538 { //------------------------- begin ----------------------
539  size_t N = last - first ;
540 
541  if ( N <= MaxThread or level < LevelThread)
542  { bs_algo::deep::intro_sort_internal ( first, last, level,comp) ;
543  return ;
544  };
545 
546  //------------------- check if sort --------------------------------------
547  bool SW = true ;
548  for ( iter_t it1 = first, it2 = first+1 ;
549  it2 != last and (SW = not comp(*it2,*it1));it1 = it2++);
550  if (SW) return ;
551 
552  //-------------------- pivoting ----------------------------------
553  pivot9 ( first, last ,comp);
554  const value_t & val = const_cast < value_t &>(* first);
555  iter_t c_first = first+1 , c_last = last-1;
556 
557  while ( comp(*c_first, val) ) ++c_first ;
558  while ( comp ( val,*c_last ) ) --c_last ;
559  while (not( c_first > c_last ))
560  { std::swap ( *(c_first++), *(c_last--));
561  while ( comp (*c_first, val) ) ++c_first;
562  while ( comp ( val, *c_last) ) --c_last ;
563  }; // End while
564 
565  std::swap ( *first , * c_last);
566  bs_tools::atomic_add ( *pf ,1);
567 
568  D.dq_mono.emplace_back (new tk_sort_t(pf,*this ,c_first, last, level-1));
569  sort_thread (pf,first, c_last, level-1);
570  return ;
571 };
