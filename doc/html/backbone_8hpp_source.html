165 template < uint32_t Block_size, class Iter_t, class Compare >
166 backbone< Block_size, Iter_t, Compare >::backbone( Iter_t first, Iter_t last,
167  Compare comp )
168  : global_range( first, last ), cmp( comp ), error( false )
169 { //------------------------- begin ---------------------------------
170  assert( ( last - first ) >= 0 );
171  if ( first == last ) return; // nothing to do
172 
173  nelem = size_t( last - first );
174  nblock = ( nelem + Block_size - 1 ) / Block_size;
175  ntail = ( nelem % Block_size );
176  index.reserve( nblock + 1 );
177 
178  for ( size_t i = 0; i < nblock; ++i ) index.emplace_back( block_pos( i ) );
179 
180  range_tail.first =
181  ( ntail == 0 ) ? last : ( first + ( ( nblock - 1 ) * Block_size ) );
182  range_tail.last = last;
