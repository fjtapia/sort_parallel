161  ::backbone(Iter_t first, Iter_t last, Compare comp)
162  : global_range(first, last), cmp(comp)
163 { //------------------------- begin ---------------------------------
164  assert((last - first) >= 0);

167  nelem = size_t(last - first);
168  nblock = (nelem + Block_size - 1) / Block_size;
169  ntail = (nelem % Block_size);
170  index.reserve(nblock + 1);

173

175  (ntail == 0) ? last : (first + ((nblock - 1) * Block_size));
176  range_tail.last = last;
