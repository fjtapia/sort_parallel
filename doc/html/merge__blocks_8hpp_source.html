129 /// @brief make the indirect merge of the two range_pos defined by their index
130 /// position [pos_index1, pos_index2 ) and [ pos_index2, pos_index3 )
131 //
132 /// @param bkb : backbone with all the data to sort , and the internal data
133 /// structures of the algorithm
134 /// @param pos_index1 : first position of the first range in the index
135 /// @param pos_index2 : last position of the first range and first position
136 /// of the second range in the index
137 /// @param pos_index3 : last position of the second range in the index
138 //-------------------------------------------------------------------------
139 template <uint32_t Block_size, uint32_t Group_size, class Iter_t, class Compare>
140 merge_blocks<Block_size, Group_size, Iter_t, Compare>::merge_blocks(
141  backbone_t &bkb, size_t pos_index1, size_t pos_index2, size_t pos_index3)
142  : bk(bkb)
143 { //----------------------------- begin --------------------------------
144  size_t nblock1 = pos_index2 - pos_index1;
145  size_t nblock2 = pos_index3 - pos_index2;
146  if (nblock1 == 0 or nblock2 == 0) return;
147 
148  //-----------------------------------------------------------------------
149  // Merging of the two intervals
150  //-----------------------------------------------------------------------
151  std::vector<block_pos> vpos1, vpos2;
152  vpos1.reserve(nblock1 + 1);
153  vpos2.reserve(nblock2 + 1);
154 
155  for (size_t i = pos_index1; i < pos_index2; ++i) {
156  vpos1.emplace_back(bk.index[i].pos(), true);
157  };
158 
159  for (size_t i = pos_index2; i < pos_index3; ++i) {
160  vpos2.emplace_back(bk.index[i].pos(), false);
161  };
162  //-------------------------------------------------------------------
163  // tail process
164  //-------------------------------------------------------------------
165  if (vpos2.back().pos() == (bk.nblock - 1) and
166  bk.range_tail.first != bk.range_tail.last)
167  {
168  tail_process(vpos1, vpos2);
169  nblock1 = vpos1.size();
170  nblock2 = vpos2.size();
171  };
172 
173  compare_block_pos_t cmp_blk(bk.global_range.first, bk.cmp);
174 
175  util::full_merge(vpos1.begin(), vpos1.end(), vpos2.begin(), vpos2.end(),
176  bk.index.begin() + pos_index1, cmp_blk);
177 
178  // Extracting the ranges for to merge the elements
179  extract_ranges(range_pos(pos_index1, pos_index1 + nblock1 + nblock2));
180 };

211  vblkpos2.emplace_back(posback1, false);

219 // function : cut_range
220 /// @brief when the rng_input is greather than Group_size, this function divide
221 /// it in several parts creating function_t elements, which are inserted
222 /// in the concurrent stack of the backbone
223 //
224 /// @param rng_input : range to divide
225 //-------------------------------------------------------------------------
226 template <uint32_t Block_size, uint32_t Group_size, class Iter_t, class Compare>
227 void merge_blocks<Block_size, Group_size, Iter_t, Compare>
228  ::cut_range(range_pos rng_input)
229 { //---------------------------- begin -----------------------------
230  if (rng_input.size() < Group_size) {
231  merge_range_pos(rng_input);
232  return;
233  };
234  atomic_t counter(0);
235  size_t npart = (rng_input.size() + Group_size - 1) / Group_size;
236  size_t size_part = rng_input.size() / npart;
237 
238  size_t pos_ini = rng_input.first;
239  size_t pos_last = rng_input.last;
240  while (pos_ini < pos_last) {
241  size_t pos = pos_ini + size_part;
242  while (pos < pos_last and
243  bk.index[pos - 1].side() == bk.index[pos].side())
244  {
245  ++pos;
246  };
247  if (pos < pos_last) {
248  in_place_merge_uncontiguous(bk.get_range(bk.index[pos - 1].pos()),
249  bk.get_range(bk.index[pos].pos()),
250  bk.get_range_buf(), bk.cmp);
251  }
252  else
253  pos = pos_last;
254  if ((pos - pos_ini) > 1) {
255  range_pos rng_aux(pos_ini, pos);
256  function_merge_range_pos(rng_aux, counter);
257  };
258  pos_ini = pos;
259  };
260  bk.exec(counter); // wait until finish all the ranges
261 };

292 // function : extract_ranges
293 /// @brief from a big range of positions of blocks in the index. Examine which
294 /// are mergeable, and generate a couple of ranges for to be merged.
295 /// With the ranges obtained generate function_t elements and are
296 /// inserted in the concurrent stack.
297 /// When the range obtained is smaller than Group_size, generate a
298 /// function_t calling to merge_range_pos, when is greater, generate a
299 /// function_t calling to cut_range
300 //
301 /// @param rpos range_input : range of the position in the index, where must
302 /// extract the ranges to merge
303 //-------------------------------------------------------------------------
304 template <uint32_t Block_size, uint32_t Group_size, class Iter_t, class Compare>
305 void merge_blocks<Block_size, Group_size, Iter_t, Compare>
306  ::extract_ranges(range_pos range_input)
307 { //------------------------------ begin -------------------------------
308  if (range_input.size() < 2) return;
309  atomic_t counter(0);
310 
311  // The names with x are positions of the index
312  size_t posx_ini = range_input.first;
313  block_pos bp_posx_ini = bk.index[posx_ini];
314 
315  range_it rng_max = bk.get_range(bp_posx_ini.pos());
316  bool side_max = bp_posx_ini.side();
317 
318  block_pos bp_posx;
319  range_it rng_posx = rng_max;
320  bool side_posx = side_max;
321 
322  for (size_t posx = posx_ini + 1; posx <= range_input.last; ++posx) {
323  bool final = (posx == range_input.last);
324  bool mergeable = false;
325 
326  if (not final) {
327  bp_posx = bk.index[posx];
328  rng_posx = bk.get_range(bp_posx.pos());
329  side_posx = bp_posx.side();
330  mergeable = (side_max != side_posx and
331  is_mergeable(rng_max, rng_posx, bk.cmp));
332  };
333 
334  if (final or not mergeable) {
335  range_pos rp_final(posx_ini, posx);
336  if (rp_final.size() > 1) {
337  if (rp_final.size() > Group_size) {
338  function_cut_range(rp_final, counter);
339  }
340  else
341  {
342  function_merge_range_pos(rp_final, counter);
343  };
344  };
345  posx_ini = posx;
346  if (not final) {
347  rng_max = rng_posx;
348  side_max = side_posx;
349  };
350  }
351  else
352  {
353  if (bk.cmp(*(rng_max.back()), *(rng_posx.back()))) {
354  rng_max = rng_posx;
355  side_max = side_posx;
356  };
357  };
358  };
359  bk.exec(counter);
360 };
