60 /// @brief : indirect sort. This function sort a vector of iterators to the
61 /// elements. And after, move the elements, for to be phisically
62 /// sorted. This is useful with big objects
63 ///
64 /// @param first : iterator to the first element of the range to sort
65 /// @param last : iterator after the last element to the range to sort
66 /// @param comp : object for to compare two elements pointed by Iter_t

69 template <class Iter_t, typename Compare = compare_iter<Iter_t>>
70 void indirect_sort(Iter_t first, Iter_t last, Compare comp = Compare())
71 {
72  typedef less_ptr_no_null<Iter_t, Compare> compare_ptr;
73 
74  std::vector<Iter_t> v_iter;
75  detail::create_index(first, last, v_iter);
76  detail::intro_sort(v_iter.begin(), v_iter.end(), compare_ptr(comp));
77  detail::sort_index(first, v_iter);
78 };

185 /// @param first : iterator to the first element of the range to sort

190 template <class Iter_t, class Compare = compare_iter<Iter_t>>
191 void indirect_stable_sort(Iter_t first, Iter_t last, Compare comp = Compare())
192 {
193  typedef less_ptr_no_null<Iter_t, Compare> compare_ptr;
194  typedef typename std::vector<Iter_t>::iterator iter_ptr;
195 
196  std::vector<Iter_t> v_iter;
197  detail::create_index(first, last, v_iter);
198  detail::spin_sort<iter_ptr, compare_ptr>(v_iter.begin(), v_iter.end(),
199  compare_ptr(comp));
200  detail::sort_index(first, v_iter);
201 };
