40 /// @brief Compare the elements pointed by it1 and it2, and if they
41 /// are equals, compare their position
42 /// @tparam compare : object to compate the elements pointed by iter1_t
43 /// @param [in] it1 : iterator to the first element
44 /// @param [in] pos1 : position of the object pointed by it1
45 /// @param [in] it2 : iterator to the second element
46 /// @param [in] pos2 : position of the element ointed by it2
47 /// @param [in] comp : comparison object
48 /// @return result of the comparison
49 //-----------------------------------------------------------------------------
50 template < class iter_t,
51  class compare = std::less< typename iterator_traits<iter_t>::value_type> >
52 inline bool less_range ( iter_t it1, uint32_t pos1 ,
53  iter_t it2, uint32_t pos2 ,compare comp =compare() )
54 { //----------------------------- begin ------------------------------------
55  if ( comp (*it1, *it2 )) return true ;
56  if ( pos2 < pos1) return false ;
57  return not ( comp (*it2, *it1 )) ;
