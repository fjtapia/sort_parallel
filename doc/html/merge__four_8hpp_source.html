47 /// @brief Compare the elements pointed by it1 and it2, and if they
48 /// are equals, compare their position, doing a stable comparison
49 ///
50 /// @param it1 : iterator to the first element
51 /// @param pos1 : position of the object pointed by it1
52 /// @param it2 : iterator to the second element
53 /// @param pos2 : position of the element pointed by it2
54 /// @param comp : comparison object
55 /// @return result of the comparison
56 //-----------------------------------------------------------------------------
57 template < class Iter_t,
58  class Compare = typename compare_iter< Iter_t >::value_type >
59 inline bool less_range (Iter_t it1, uint32_t pos1, Iter_t it2, uint32_t pos2,
60  Compare comp = Compare ( ))
61 {
62  return (comp (*it1, *it2)) ? true : (pos2 < pos1) ? false
