359 // function : merge_flow
360 /// @brief : merge two ranges, as part of a merge the ranges in a list. This
361 /// function reduce the number of movements compared with inplace_merge
362 /// when you need to merge a sequence of ranges.
363 /// This function merge the ranges rbuf and rng2, and the results
364 /// are in rng1 and rbuf
365 //
366 /// @param rng1 : range where locate the first elements of the merge
367 /// @param rbuf : range which provide the first elements, and where store
368 /// the last results of the merge
369 /// @param rng2 : range which provide the last elements to merge
370 /// @param comp : object for to compare elements
371 /// @return true : not changes done, false : changes in the buffers
372 //-----------------------------------------------------------------------------
373 template <class Iter1_t, class Iter2_t, class Compare>
374 void merge_flow(range<Iter1_t> rng1,
375  range<Iter2_t> rbuf,
376  range<Iter1_t> rng2, Compare cmp)
377 { //------------------- metaprogramming -----------------------------------
378  typedef typename iterator_traits<Iter1_t>::value_type type1;
379  typedef typename iterator_traits<Iter2_t>::value_type type2;
380  static_assert(std::is_same<type1, type2>::value,
381  "Incompatible iterators\n");
382  //---------------------------- begin --------------------------------------
383  range<Iter2_t> rbx(rbuf);
384  range<Iter1_t> rx1(rng1), rx2(rng2);
385  assert(rbx.size() == rx1.size() and rx1.size() == rx2.size());
386  while (rx1.first != rx1.last) {
387  *(rx1.first++) = (cmp(*rbx.first, *rx2.first))
388  ? std::move(*(rbx.first++))
389  : std::move(*(rx2.first++));
390  };
391  if (rx2.first == rx2.last) return;
392  if (rbx.first == rbx.last)
393  util::init_move(rbuf, rng2);
394  else
395  util::half_merge(rbuf, rx2, rbx, cmp);
396 };
