60 ///

62 /// @param last : iterator to the last element to initialize
63 /// @param val : object used for the initialization

65 template <class Iter_t>
66 void init(Iter_t first, Iter_t last,
67  typename iterator_traits<Iter_t>::value_type &val)
68 { //----------------- begin ---------------------------
69  if (first == last) return;
70  construct(&(*first), std::move(val));

144  const Iter1_t end_buf2, Iter2_t buf_out, Compare comp)

152  while ((buf1 != end_buf1) and (buf2 != end_buf2)) {

245  typedef typename iterator_traits<Iter2_t>::value_type type2;
246  typedef typename iterator_traits<Iter3_t>::value_type type3;
247  static_assert(std::is_same<type1, type2>::value,

250  "Incompatible iterators\n");

252  //--------------------- code -------------------------------------------
253  if (src1 == end_src1 or src2 == end_src2 or
254  not comp(*src2, *(end_src1 - 1)))
255  return true;

265  }

269  init_move(src2_first, aux, end_aux);
