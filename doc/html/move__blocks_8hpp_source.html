163 
164  size_t pos_index_ini = 0, pos_index_src = 0, pos_index_dest = 0;
165  while ( pos_index_ini < bk.index.size() ) {
166  while ( pos_index_ini < bk.index.size() and
167  bk.index[ pos_index_ini ].pos() == pos_index_ini )

170  };
171 
172  if ( pos_index_ini == bk.index.size() ) break;
173 
174  sequence.clear();
175  pos_index_src = pos_index_dest = pos_index_ini;
176  sequence.push_back( pos_index_ini );
177

179  pos_index_src = bk.index[ pos_index_dest ].pos();
180  sequence.push_back( pos_index_src );
181 
182  bk.index[ pos_index_dest ].set_pos( pos_index_dest );
183  pos_index_dest = pos_index_src;
184  };

187  vsequence.push_back( sequence );

191  }

229 /// @brief move the blocks, following the positions of the init_sequence.

238  class Compare >
239 void move_blocks< Block_size, Group_size, Iter_t, Compare >::move_long_sequence(
240  const std::vector< size_t > &init_sequence )
241 { //--------------------------------- begin ----------------------------
242  if ( init_sequence.size() < Group_size )
243  return move_sequence( init_sequence );
244 
245  size_t npart = ( init_sequence.size() + Group_size - 1 ) / Group_size;
246  size_t size_part = init_sequence.size() / npart;
247  atomic_t son_counter( 0 );
248 
249  std::vector< size_t > sequence;
250  sequence.reserve( size_part );
251 
252  std::vector< size_t > index_seq;
253  index_seq.reserve( npart );
254 
255  auto it_pos = init_sequence.begin();
256  for ( size_t i = 0; i < ( npart - 1 ); ++i, it_pos += size_part ) {
257  sequence.assign( it_pos, it_pos + size_part );
258  index_seq.emplace_back( *( it_pos + size_part - 1 ) );
259  function_move_sequence( sequence, son_counter, bk.error );
260  };
261 
262  sequence.assign( it_pos, init_sequence.end() );
263  index_seq.emplace_back( init_sequence.back() );
264  function_move_sequence( sequence, son_counter, bk.error );
265 
266  bk.exec( son_counter );
267  if ( bk.error ) return;
268  move_long_sequence( index_seq );
