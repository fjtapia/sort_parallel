116  :comp(comp1)
117 { //------------------------- begin ----------------------
118  size_t N = last - first;
119  assert ( N >=0);
120 
121  uint32_t NBSize = (NBits64(sizeof (value_t)));
122  if ( NBSize > 10 ) NBSize = 10 ;
123  MaxPerThread = 1<< (19 - NBSize);
124 
125  uint32_t Level = (NBits64 ( N) <<1);
126  LevelThread = (NBits64 ( MaxPerThread) <<1);
127 
128  if ( (size_t)N < MaxPerThread or NT() == 1 )
129  { intro_sort ( first, last, comp) ;
130  return ;
131  } ;
132  //------------------- check if sort ----------------------------------
133  bool SW = true ;
134  for ( iter_t it1 = first, it2 = first+1 ;
135  it2 != last and (SW = not comp(*it2,*it1));it1 = it2++);
136  if (SW) return ;

138  //-------------------------------------------------------------------
139  ST.reserve ( Level * NT() );
140  std::atomic_store (&ND, N);
141  ST.emplace_back ( first, last, Level);

155 template < class iter_t, typename compare >
156 void parallel_sort_comp<iter_t,compare>::sort_thread( )
157 { //------------------------- begin ----------------------
158  token_t tk ;
159  while ( atomic_read ( ND) != 0)
160  { //--------------------------------------------------------------
161  // read from ST
162  //--------------------------------------------------------------
163  if ( not ST.pop_copy_back( tk) )
164  { std::this_thread::yield() ;
165  continue ;
166  };
167  size_t N = tk.size() ;
168  if ( N <= MaxPerThread or tk.level < LevelThread)
169  { atomic_sub ( ND, N );
170  deep::intro_sort_internal ( tk.first, tk.last, tk.level,comp) ;
171  continue ;
172  };
173 
174  //----------------------------------------------------------------
175  // split
176  //----------------------------------------------------------------
177  typedef typename iterator_traits<iter_t>::value_type value_t ;
178 
179  //------------------- check if sort ------------------------------
180  bool SW = true ;
181  for ( iter_t it1 = tk.first, it2 = tk.first+1 ;
182  it2 != tk.last and (SW = not comp(*it2,*it1));it1 = it2++);
183  if (SW)
184  { atomic_sub ( ND, N );
185  continue ;
186  };
187  //---------------------- pivot select ----------------------------
188  deep::pivot9 ( tk.first, tk.last , comp) ;
189  value_t & val = const_cast < value_t &>(* tk.first);
190  iter_t c_first = tk.first+1 , c_last = tk.last-1;
191 
192  while ( c_first != tk.last and comp (*c_first, val)) ++c_first ;
193  while ( comp(val ,*c_last ) ) --c_last ;
194  while (not( c_first > c_last ))
195  { std::swap ( *(c_first++), *(c_last--));
196  while ( comp(*c_first,val) ) ++c_first;
197  while ( comp(val, *c_last) ) --c_last ;
198  }; // End while
199  std::swap ( *tk.first , *c_last);
200 
201  //----------------------------------------------------------------
202  // Store the tokens en ST
203  //----------------------------------------------------------------
204  token_t tk1 ( tk.first, c_last, tk.level-1) ;
205  token_t tk2 ( c_first , tk.last, tk.level-1) ;
206  auto N1 = (tk1.last - tk1.first);
207  auto N2 = (tk2.last - tk2.first);
208  auto N3 = N - N1- N2 ;
209  if ( N3 != 0) atomic_sub ( ND, N3 );
210 
211  ST.push_back ( tk2);
212  ST.push_back ( tk1);
213 
214  }; // end while
215 } ;
